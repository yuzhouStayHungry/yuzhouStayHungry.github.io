class Solution:
    def twoSum(self, nums: List[int], target: int) -> List[int]:

        rPointer, lPointer = 0, len(nums) - 1
        while rPointer < lPointer:

            twoSum = nums[rPointer] + nums[lPointer]
            if twoSum == target:
                return nums[rPointer], nums[lPointer]
            elif twoSum > target:
                lPointer -= 1
            else:
                rPointer += 1
        return []
